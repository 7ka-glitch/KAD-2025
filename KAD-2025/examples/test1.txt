main
{
    write("--- Math Test ---");
    
    int a;
    a = 2 + 2 * 2;      
    write(a);           # Ожидается: 6
    
    a = (2 + 2) * 2;
    write(a);           # Ожидается: 8
    
    int b;
    b = 10 / 2 + 5;     
    write(b);          # Ожидается: 10 (деление раньше сложения)

    # Тест на степень (справа налево или слева направо)
    # Обычно 2^3^2 = 2^(9) = 512 Если будет (2^3)^2 = 64, значит ассоциативность левая
    int c;
    c = 2^3^2;
    write(c);
}
